INCLUDES E REPEAT 
---------------------------------------------------------------------------------------------------------

includes serve para string e array
retorna true ou false

array

let lista = [ 'café, 'arroz', 'ovo', 'feijao'];


// para saber se dentro dessa lista tem por exemplo o item 'carne'

console.log(lista.includes('carne'));
